    <script>
        function updateCategoria(nivel) {
            var cat = document.getElementById('categoria');
            cat.innerHTML = '';
            if (nivel === 'Profissional') {
                cat.innerHTML = '<option>Não se aplica</option>';
                cat.disabled = true;
            } else if (nivel === 'Amadora') {
                cat.innerHTML =
                    '<option value="">Selecione</option>' +
                    '<option value="A">A</option>' +
                    '<option value="B">B</option>' +
                    '<option value="C">C</option>' +
                    '<option value="D">D</option>' +
                    '<option value="40+">40+</option>' +
                    '<option value="50+">50+</option>' +
                    '<option value="60+">60+</option>';
                cat.disabled = false;
            } else {
                cat.innerHTML = '<option>Selecione o nível primeiro</option>';
                cat.disabled = true;
            }
        }
        function maskCPF(field) {
            field.value = field.value.replace(/\D/g,'')
                .replace(/(\d{3})(\d)/,'$1.$2')
                .replace(/(\d{3})(\d)/,'$1.$2')
                .replace(/(\d{3})(\d{1,2})/,'$1-$2').substr(0,14);
        }
        function maskTelefone(field) {
            field.value = field.value.replace(/\D/g,'')
                .replace(/(\d{2})(\d)/,'($1) $2')
                .replace(/(\d{5})(\d)/,'$1-$2').substr(0,15);
        }
        function maskCEP(field) {
            field.value = field.value.replace(/\D/g,'')
                .replace(/(\d{5})(\d)/,'$1-$2').substr(0,9);
        }
